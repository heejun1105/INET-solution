# INET 프로젝트 환경변수 설정 예시 파일
#
# 사용 방법:
# 1. 이 파일을 복사해서 .env 로 이름 변경
# 2. 아래 값들을 실제 비밀번호로 변경
# 3. docker-compose.yml은 수정하지 않아도 됩니다!
#
# ⚠️ 중요: .env 파일은 절대 Git에 커밋하지 마세요!

# MySQL 루트 비밀번호 (관리자용)
MYSQL_ROOT_PASSWORD=root_password_변경_필요

# MySQL 사용자 이름
MYSQL_USER=inet_user

# MySQL 사용자 비밀번호 (애플리케이션 연결용)
# ⚠️ 주의: 이 값은 SPRING_DATASOURCE_PASSWORD와 동일해야 합니다!
MYSQL_PASSWORD=db_password_변경_필요
